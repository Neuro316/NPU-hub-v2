'use client'

import { useState, useEffect, useCallback, useMemo, useRef } from 'react'
import { useWorkspace } from '@/lib/workspace-context'
import { usePodcastData } from '@/lib/hooks/use-podcast-data'
import type { Podcast, PodcastQuestion, PodcastFutureIdea, AdvisoryVoice, PodcastChecklist } from '@/lib/types/podcast'
import { PODCAST_STATUS, PODCAST_FORMAT, PODCAST_PLATFORMS, DEFAULT_ICPS } from '@/lib/types/podcast'
import {
  Mic, Plus, X, ChevronDown, ChevronUp, Search, Sparkles, Bot, Loader2,
  Calendar, Clock, User, Globe, FileText, MessageSquare, Target, Lightbulb,
  CheckSquare, Square, Trash2, Upload, Play, Pause, Send, Volume2, Star,
  ExternalLink, Copy, Pin, Settings2, Users, Zap, BookOpen, ArrowRight,
  BarChart3, TrendingUp, AlertCircle
} from 'lucide-react'

/* ═══════════════════════════════════════════
   Collapsible Section Component
   ═══════════════════════════════════════════ */
function Section({ title, icon, children, defaultOpen = false, badge, color }: {
  title: string, icon: React.ReactNode, children: React.ReactNode,
  defaultOpen?: boolean, badge?: string | number, color?: string
}) {
  const [open, setOpen] = useState(defaultOpen)
  return (
    <div className="border border-gray-100 rounded-xl overflow-hidden mb-3">
      <button onClick={() => setOpen(!open)}
        className="w-full flex items-center gap-2.5 px-4 py-3 bg-gray-50/50 hover:bg-gray-50 transition-colors text-left">
        <span className="text-sm">{icon}</span>
        <span className="text-xs font-semibold text-np-dark flex-1">{title}</span>
        {badge !== undefined && (
          <span className="text-[9px] font-bold px-1.5 py-0.5 rounded-full bg-np-blue/10 text-np-blue">{badge}</span>
        )}
        {open ? <ChevronUp className="w-3.5 h-3.5 text-gray-400" /> : <ChevronDown className="w-3.5 h-3.5 text-gray-400" />}
      </button>
      {open && <div className="px-4 py-3 bg-white">{children}</div>}
    </div>
  )
}

/* ═══════════════════════════════════════════
   Field Components
   ═══════════════════════════════════════════ */
function Field({ label, icon, children }: { label: string, icon?: string, children: React.ReactNode }) {
  return (
    <div className="mb-3">
      <label className="text-[10px] font-bold text-gray-400 uppercase tracking-wider mb-1 flex items-center gap-1">
        {icon && <span>{icon}</span>} {label}
      </label>
      {children}
    </div>
  )
}

const inputCls = "w-full text-xs border border-gray-200 rounded-lg px-3 py-2 focus:outline-none focus:ring-2 focus:ring-np-blue/20 placeholder-gray-300"
const textareaCls = "w-full text-xs border border-gray-200 rounded-lg px-3 py-2 focus:outline-none focus:ring-2 focus:ring-np-blue/20 placeholder-gray-300 resize-none"
const selectCls = "w-full text-xs border border-gray-200 rounded-lg px-3 py-2 focus:outline-none focus:ring-1 focus:ring-np-blue/30"

/* ═══════════════════════════════════════════
   Main Page
   ═══════════════════════════════════════════ */
export default function MediaAppearancesPage() {
  const { currentOrg, user, loading: orgLoading } = useWorkspace()
  const {
    podcasts, futureIdeas, advisoryVoices, loading,
    addPodcast, updatePodcast, deletePodcast,
    fetchQuestions, addQuestion, updateQuestion, deleteQuestion, bulkAddQuestions,
    fetchChecklist, toggleChecklistItem, addChecklistItem,
    addFutureIdea, updateFutureIdea, deleteFutureIdea,
    addVoice, updateVoice, deleteVoice, toggleVoice,
  } = usePodcastData()

  // ─── State ───
  const [modalOpen, setModalOpen] = useState(false)
  const [editingPodcast, setEditingPodcast] = useState<Podcast | null>(null)
  const [filterStatus, setFilterStatus] = useState<string>('all')
  const [searchQuery, setSearchQuery] = useState('')

  // ─── Modal form state ───
  const [form, setForm] = useState<Partial<Podcast>>({})
  const [questions, setQuestions] = useState<PodcastQuestion[]>([])
  const [checklist, setChecklist] = useState<PodcastChecklist[]>([])
  const [newQuestion, setNewQuestion] = useState('')
  const [newCheckItem, setNewCheckItem] = useState('')
  const [newIdeaTitle, setNewIdeaTitle] = useState('')
  const [newTopicTag, setNewTopicTag] = useState('')
  const [saving, setSaving] = useState(false)

  // ─── AI state ───
  const [aiFilling, setAiFilling] = useState(false)
  const [aiDrafting, setAiDrafting] = useState(false)
  const [feedbackLoading, setFeedbackLoading] = useState(false)
  const [feedbackResults, setFeedbackResults] = useState<any>(null)
  const [socialPosts, setSocialPosts] = useState<any[]>([])
  const [socialLoading, setSocialLoading] = useState(false)

  // ─── Voice panel state ───
  const [voicePanelOpen, setVoicePanelOpen] = useState(false)
  const [newVoiceName, setNewVoiceName] = useState('')
  const [newVoiceRole, setNewVoiceRole] = useState('')
  const [newVoicePerspective, setNewVoicePerspective] = useState('')

  // ─── Prep sheet upload ───
  const [prepSheetDragging, setPrepSheetDragging] = useState(false)
  const [parsingPrepSheet, setParsingPrepSheet] = useState(false)
  const fileInputRef = useRef<HTMLInputElement>(null)

  // ─── Voice-to-text ───
  const [isRecording, setIsRecording] = useState(false)
  const [recordingField, setRecordingField] = useState<string | null>(null)
  const mediaRecorderRef = useRef<MediaRecorder | null>(null)

  const updateForm = (key: string, value: any) => setForm(prev => ({ ...prev, [key]: value }))

  // ─── Open modal ───
  const openModal = async (podcast?: Podcast) => {
    if (podcast) {
      setEditingPodcast(podcast)
      setForm(podcast)
      const [qs, cl] = await Promise.all([
        fetchQuestions(podcast.id),
        fetchChecklist(podcast.id),
      ])
      setQuestions(qs)
      setChecklist(cl)
    } else {
      setEditingPodcast(null)
      setForm({ status: 'upcoming', format: 'interview', platform: 'Zoom', crossover_topics: [], target_icps: [] })
      setQuestions([])
      setChecklist([])
    }
    setFeedbackResults(null)
    setSocialPosts([])
    setModalOpen(true)
  }

  // ─── Save podcast ───
  const handleSave = async () => {
    if (!form.name?.trim()) return
    setSaving(true)
    try {
      if (editingPodcast) {
        await updatePodcast(editingPodcast.id, form)
      } else {
        await addPodcast(form)
      }
      setModalOpen(false)
    } catch (e) { console.error(e) }
    setSaving(false)
  }

  // ─── AI Fill ───
  const handleAIFill = async () => {
    if (!form.name?.trim()) return
    setAiFilling(true)
    try {
      const res = await fetch('/api/ai/podcast', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ action: 'ai_fill', data: form }),
      })
      const data = await res.json()
      if (!data.error) {
        const updates: any = {}
        const fieldMap = ['episode_topic', 'host_name', 'host_excited_about', 'host_cares_about',
          'interview_style', 'show_website', 'audience_info', 'show_notes', 'strategic_positioning']
        fieldMap.forEach(f => { if (data[f]) updates[f] = data[f] })
        if (data.suggested_talking_points) updates.key_talking_points = data.suggested_talking_points
        setForm(prev => ({ ...prev, ...updates }))
        // Add suggested questions
        if (data.suggested_questions?.length && editingPodcast) {
          const added = await bulkAddQuestions(editingPodcast.id, data.suggested_questions, 'ai')
          setQuestions(prev => [...prev, ...added])
        }
      }
    } catch (e) { console.error(e) }
    setAiFilling(false)
  }

  // ─── Parse Prep Sheet ───
  const handlePrepSheet = async (text: string) => {
    setParsingPrepSheet(true)
    try {
      const res = await fetch('/api/ai/podcast', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ action: 'parse_prep_sheet', data: { text } }),
      })
      const data = await res.json()
      if (!data.error) {
        const updates: any = { prep_sheet_text: text }
        const fields = ['name', 'episode_topic', 'host_name', 'host_email', 'show_website',
          'audience_info', 'interview_style', 'host_excited_about', 'host_cares_about',
          'show_notes', 'key_talking_points', 'platform']
        fields.forEach(f => { if (data[f] && !form[f as keyof typeof form]) updates[f] = data[f] })
        if (data.recording_date) updates.recording_date = data.recording_date
        setForm(prev => ({ ...prev, ...updates }))
        if (data.suggested_questions?.length && editingPodcast) {
          const added = await bulkAddQuestions(editingPodcast.id, data.suggested_questions, 'prep_sheet')
          setQuestions(prev => [...prev, ...added])
        }
      }
    } catch (e) { console.error(e) }
    setParsingPrepSheet(false)
  }

  const handleFileUpload = async (file: File) => {
    const text = await file.text()
    handlePrepSheet(text)
  }

  // ─── Feedback Coach ───
  const handleFeedbackCoach = async () => {
    const activeVoices = advisoryVoices.filter(v => v.active)
    if (activeVoices.length === 0) return
    setFeedbackLoading(true)
    try {
      const res = await fetch('/api/ai/podcast', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          action: 'feedback_coach',
          data: { ...form, voices: activeVoices },
        }),
      })
      const data = await res.json()
      if (!data.error) setFeedbackResults(data)
    } catch (e) { console.error(e) }
    setFeedbackLoading(false)
  }

  // ─── Draft Answers ───
  const handleDraftAnswers = async () => {
    if (questions.length === 0) return
    setAiDrafting(true)
    try {
      const res = await fetch('/api/ai/podcast', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          action: 'draft_answers',
          data: { ...form, questions: questions.map(q => q.question) },
        }),
      })
      const data = await res.json()
      if (data.answers) {
        for (const a of data.answers) {
          const q = questions.find(q => q.question === a.question)
          if (q) {
            await updateQuestion(q.id, { draft_answer: a.answer })
            setQuestions(prev => prev.map(qq => qq.id === q.id ? { ...qq, draft_answer: a.answer } : qq))
          }
        }
      }
    } catch (e) { console.error(e) }
    setAiDrafting(false)
  }

  // ─── Generate Social Posts ───
  const handleGenerateSocial = async () => {
    setSocialLoading(true)
    try {
      const res = await fetch('/api/ai/podcast', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ action: 'generate_social', data: form }),
      })
      const data = await res.json()
      if (data.posts) setSocialPosts(data.posts)
    } catch (e) { console.error(e) }
    setSocialLoading(false)
  }

  // ─── Voice recording (Web Speech API) ───
  const startRecording = (field: string) => {
    if (!('webkitSpeechRecognition' in window || 'SpeechRecognition' in window)) {
      alert('Speech recognition not supported in this browser')
      return
    }
    const SpeechRecognition = (window as any).webkitSpeechRecognition || (window as any).SpeechRecognition
    const recognition = new SpeechRecognition()
    recognition.continuous = true
    recognition.interimResults = true
    recognition.lang = 'en-US'
    recognition.onresult = (event: any) => {
      let transcript = ''
      for (let i = 0; i < event.results.length; i++) {
        transcript += event.results[i][0].transcript
      }
      updateForm(field, (form[field as keyof typeof form] || '') + ' ' + transcript)
    }
    recognition.onend = () => setIsRecording(false)
    recognition.start()
    setIsRecording(true)
    setRecordingField(field)
    mediaRecorderRef.current = recognition
  }

  const stopRecording = () => {
    if (mediaRecorderRef.current) {
      (mediaRecorderRef.current as any).stop()
    }
    setIsRecording(false)
    setRecordingField(null)
  }

  // ─── Add question ───
  const handleAddQuestion = async () => {
    if (!newQuestion.trim() || !editingPodcast) return
    const q = await addQuestion(editingPodcast.id, newQuestion.trim())
    if (q) setQuestions(prev => [...prev, q])
    setNewQuestion('')
  }

  // ─── Toggle ICP ───
  const toggleICP = (icp: string) => {
    const current = form.target_icps || []
    const updated = current.includes(icp)
      ? current.filter(i => i !== icp)
      : [...current, icp]
    updateForm('target_icps', updated)
  }

  // ─── Add crossover topic ───
  const addTopicTag = () => {
    if (!newTopicTag.trim()) return
    const current = form.crossover_topics || []
    if (!current.includes(newTopicTag.trim())) {
      updateForm('crossover_topics', [...current, newTopicTag.trim()])
    }
    setNewTopicTag('')
  }

  // ─── Filtered podcasts ───
  const filteredPodcasts = useMemo(() => {
    let list = podcasts
    if (filterStatus !== 'all') list = list.filter(p => p.status === filterStatus)
    if (searchQuery.trim()) {
      const q = searchQuery.toLowerCase()
      list = list.filter(p =>
        p.name.toLowerCase().includes(q) || p.host_name?.toLowerCase().includes(q) ||
        p.episode_topic?.toLowerCase().includes(q)
      )
    }
    return list
  }, [podcasts, filterStatus, searchQuery])

  // ─── Stats ───
  const stats = useMemo(() => ({
    total: podcasts.length,
    upcoming: podcasts.filter(p => p.status === 'upcoming').length,
    ready: podcasts.filter(p => p.status === 'ready').length,
    completed: podcasts.filter(p => p.status === 'completed').length,
  }), [podcasts])

  if (orgLoading || loading) {
    return <div className="flex items-center justify-center h-64"><div className="animate-pulse text-gray-400">Loading...</div></div>
  }

  return (
    <div>
      {/* ═══ Header ═══ */}
      <div className="flex items-center gap-3 mb-4">
        <div className="mr-2">
          <h1 className="text-lg font-bold text-np-dark">Media Appearances</h1>
          <p className="text-[10px] text-gray-400">{currentOrg?.name} · Podcasts, Interviews & Press</p>
        </div>

        <select value={filterStatus} onChange={e => setFilterStatus(e.target.value)}
          className="h-9 px-3 rounded-lg border border-gray-200 text-xs font-medium text-np-dark">
          <option value="all">All ({stats.total})</option>
          <option value="upcoming">🟢 Upcoming ({stats.upcoming})</option>
          <option value="prep_needed">🟡 Prep Needed</option>
          <option value="ready">✅ Ready ({stats.ready})</option>
          <option value="completed">⬛ Completed ({stats.completed})</option>
        </select>

        <div className="flex items-center gap-1.5 h-9 bg-white border border-gray-200 rounded-lg px-3 flex-1 max-w-xs">
          <Search className="w-3.5 h-3.5 text-gray-400" />
          <input value={searchQuery} onChange={e => setSearchQuery(e.target.value)}
            placeholder="Search podcasts..." className="flex-1 text-xs bg-transparent focus:outline-none placeholder-gray-300" />
        </div>

        <div className="flex-1" />

        <button onClick={() => setVoicePanelOpen(true)}
          className="flex items-center gap-1.5 h-9 px-3 rounded-lg text-xs font-medium text-gray-500 hover:text-gray-700 hover:bg-gray-50 transition-all">
          <Users className="w-3.5 h-3.5" /> Voices
        </button>

        <button onClick={() => openModal()}
          className="flex items-center gap-1.5 h-9 px-3.5 rounded-lg bg-np-blue text-white text-xs font-medium hover:bg-np-blue/90 shadow-sm shadow-np-blue/20">
          <Plus className="w-3.5 h-3.5" /> New Podcast
        </button>
      </div>

      {/* ═══ Stats Row ═══ */}
      <div className="grid grid-cols-4 gap-3 mb-4">
        {[
          { label: 'Total', value: stats.total, color: '#3B82F6' },
          { label: 'Upcoming', value: stats.upcoming, color: '#10B981' },
          { label: 'Ready', value: stats.ready, color: '#6366F1' },
          { label: 'Completed', value: stats.completed, color: '#6B7280' },
        ].map(s => (
          <div key={s.label} className="bg-white border border-gray-100 rounded-xl p-4 text-center">
            <p className="text-2xl font-bold" style={{ color: s.color }}>{s.value}</p>
            <p className="text-[10px] text-gray-400 font-medium uppercase tracking-wider mt-0.5">{s.label}</p>
          </div>
        ))}
      </div>

      {/* ═══ Podcast Cards ═══ */}
      {filteredPodcasts.length === 0 ? (
        <div className="bg-white border border-gray-100 rounded-2xl p-16 text-center">
          <Mic className="w-14 h-14 text-gray-200 mx-auto mb-4" />
          <h2 className="text-lg font-semibold text-np-dark mb-2">No Appearances Yet</h2>
          <p className="text-sm text-gray-500 mb-6 max-w-md mx-auto">
            Track podcast appearances, prep talking points, get AI feedback from your advisory panel.
          </p>
          <button onClick={() => openModal()} className="btn-primary">Add First Podcast</button>
        </div>
      ) : (
        <div className="space-y-2 mb-6">
          {filteredPodcasts.map(pod => (
            <button key={pod.id} onClick={() => openModal(pod)}
              className="w-full bg-white border border-gray-100 rounded-xl p-4 hover:border-gray-200 hover:shadow-sm transition-all text-left group">
              <div className="flex items-start gap-3">
                <Mic className="w-5 h-5 text-gray-300 mt-0.5 flex-shrink-0" />
                <div className="flex-1 min-w-0">
                  <div className="flex items-center gap-2 mb-1">
                    <span className="font-semibold text-sm text-np-dark">{pod.name}</span>
                    <span className="text-[10px] px-2 py-0.5 rounded-full font-medium"
                      style={{ backgroundColor: PODCAST_STATUS[pod.status].color + '15', color: PODCAST_STATUS[pod.status].color }}>
                      {PODCAST_STATUS[pod.status].icon} {PODCAST_STATUS[pod.status].label}
                    </span>
                  </div>
                  {pod.episode_topic && <p className="text-xs text-gray-500 mb-1">{pod.episode_topic}</p>}
                  <div className="flex items-center gap-3 text-[10px] text-gray-400 flex-wrap">
                    {pod.host_name && <span className="flex items-center gap-1"><User className="w-3 h-3" />{pod.host_name}</span>}
                    {pod.recording_date && <span className="flex items-center gap-1"><Calendar className="w-3 h-3" />{new Date(pod.recording_date).toLocaleDateString()}</span>}
                    {pod.format && <span>{PODCAST_FORMAT[pod.format].icon} {PODCAST_FORMAT[pod.format].label}</span>}
                  </div>
                  {pod.target_icps?.length > 0 && (
                    <div className="flex gap-1 mt-2 flex-wrap">
                      {pod.target_icps.map(icp => (
                        <span key={icp} className="text-[9px] px-1.5 py-0.5 rounded bg-np-blue/5 text-np-blue font-medium">{icp}</span>
                      ))}
                    </div>
                  )}
                </div>
              </div>
            </button>
          ))}
        </div>
      )}

      {/* ═══ Future Ideas Section ═══ */}
      <Section title="Future Podcast Ideas" icon={<Pin className="w-4 h-4 text-amber-500" />}
        badge={futureIdeas.length} defaultOpen={futureIdeas.length > 0}>
        <p className="text-[10px] text-gray-400 mb-3">
          Capture topics from retrospects and brainstorming. Use these to plan future episodes and sharpen your message.
        </p>
        <div className="flex gap-2 mb-3">
          <input value={newIdeaTitle} onChange={e => setNewIdeaTitle(e.target.value)}
            onKeyDown={e => { if (e.key === 'Enter' && newIdeaTitle.trim()) { addFutureIdea(newIdeaTitle.trim()); setNewIdeaTitle('') } }}
            placeholder="Quick-add an idea..." className={`${inputCls} flex-1`} />
          <button onClick={() => { if (newIdeaTitle.trim()) { addFutureIdea(newIdeaTitle.trim()); setNewIdeaTitle('') } }}
            className="px-3 py-2 bg-np-blue text-white rounded-lg text-[10px] font-medium">+ Add</button>
        </div>
        <div className="space-y-1">
          {futureIdeas.map(idea => (
            <div key={idea.id} className="flex items-center gap-2 px-3 py-2 rounded-lg border border-gray-100 group">
              <Pin className="w-3 h-3 text-amber-400 flex-shrink-0" />
              <span className="text-xs text-np-dark flex-1">{idea.title}</span>
              <select value={idea.status}
                onChange={e => updateFutureIdea(idea.id, { status: e.target.value as any })}
                className="text-[9px] border border-gray-200 rounded px-1.5 py-0.5">
                <option value="new">New</option>
                <option value="developing">Developing</option>
                <option value="ready">Ready</option>
                <option value="used">Used</option>
              </select>
              <button onClick={() => deleteFutureIdea(idea.id)}
                className="opacity-0 group-hover:opacity-100 p-1 hover:bg-red-50 rounded transition-all">
                <Trash2 className="w-3 h-3 text-gray-300 hover:text-red-500" />
              </button>
            </div>
          ))}
        </div>
      </Section>

      {/* ═══════════════════════════════════════════
          PODCAST MODAL
          ═══════════════════════════════════════════ */}
      {modalOpen && (
        <div className="fixed inset-0 bg-black/40 z-50 flex items-start justify-center pt-8 pb-8 overflow-y-auto">
          <div className="bg-white rounded-2xl w-full max-w-2xl shadow-2xl mx-4">
            {/* Modal header */}
            <div className="flex items-center gap-3 px-6 py-4 border-b border-gray-100 sticky top-0 bg-white rounded-t-2xl z-10">
              <Mic className="w-5 h-5 text-np-blue" />
              <h2 className="text-sm font-bold text-np-dark flex-1">
                {editingPodcast ? form.name || 'Edit Podcast' : 'New Podcast'}
              </h2>
              <button onClick={handleAIFill} disabled={aiFilling || !form.name}
                className="flex items-center gap-1.5 px-3 py-1.5 bg-gradient-to-r from-np-blue to-indigo-500 text-white rounded-lg text-[10px] font-medium hover:opacity-90 disabled:opacity-50">
                {aiFilling ? <Loader2 className="w-3 h-3 animate-spin" /> : <Sparkles className="w-3 h-3" />}
                AI Fill
              </button>
              <button onClick={() => setModalOpen(false)} className="p-1.5 hover:bg-gray-100 rounded-lg">
                <X className="w-4 h-4 text-gray-400" />
              </button>
            </div>

            <div className="px-6 py-4 max-h-[75vh] overflow-y-auto space-y-0">

              {/* ─── Core Fields ─── */}
              <Field label="Podcast Name" icon="🎙️">
                <input value={form.name || ''} onChange={e => updateForm('name', e.target.value)}
                  placeholder="e.g. Parent Pod" className={inputCls} />
              </Field>

              <Field label="Episode Topic / Title" icon="📝">
                <input value={form.episode_topic || ''} onChange={e => updateForm('episode_topic', e.target.value)}
                  placeholder="Topic you'll discuss" className={inputCls} />
              </Field>

              <div className="grid grid-cols-2 gap-3">
                <Field label="Recording Date" icon="📅">
                  <input type="datetime-local" value={form.recording_date?.slice(0, 16) || ''}
                    onChange={e => updateForm('recording_date', e.target.value)} className={inputCls} />
                </Field>
                <Field label="Release Date" icon="🚀">
                  <input type="date" value={form.release_date || ''}
                    onChange={e => updateForm('release_date', e.target.value)} className={inputCls} />
                </Field>
              </div>

              <div className="grid grid-cols-2 gap-3">
                <Field label="Status">
                  <select value={form.status || 'upcoming'} onChange={e => updateForm('status', e.target.value)} className={selectCls}>
                    {Object.entries(PODCAST_STATUS).map(([k, v]) => (
                      <option key={k} value={k}>{v.icon} {v.label}</option>
                    ))}
                  </select>
                </Field>
                <Field label="Format">
                  <select value={form.format || 'interview'} onChange={e => updateForm('format', e.target.value)} className={selectCls}>
                    {Object.entries(PODCAST_FORMAT).map(([k, v]) => (
                      <option key={k} value={k}>{v.icon} {v.label}</option>
                    ))}
                  </select>
                </Field>
              </div>

              <Field label="Recording Platform">
                <select value={form.platform || 'Zoom'} onChange={e => updateForm('platform', e.target.value)} className={selectCls}>
                  {PODCAST_PLATFORMS.map(p => <option key={p} value={p}>{p}</option>)}
                </select>
              </Field>

              {/* ─── Tags ─── */}
              <Field label="Crossover Topics" icon="🏷️">
                <div className="flex gap-1.5 flex-wrap mb-2">
                  {(form.crossover_topics || []).map(t => (
                    <span key={t} className="text-[10px] px-2 py-1 rounded-md bg-gray-100 text-np-dark flex items-center gap-1">
                      {t}
                      <button onClick={() => updateForm('crossover_topics', (form.crossover_topics || []).filter(x => x !== t))}>
                        <X className="w-2.5 h-2.5 text-gray-400" />
                      </button>
                    </span>
                  ))}
                </div>
                <div className="flex gap-1.5">
                  <input value={newTopicTag} onChange={e => setNewTopicTag(e.target.value)}
                    onKeyDown={e => { if (e.key === 'Enter') addTopicTag() }}
                    placeholder="Type + Enter..." className={`${inputCls} flex-1`} />
                </div>
                <p className="text-[9px] text-gray-300 mt-1">Topics that span multiple episodes (e.g. HRV, Capacity, Co-regulation)</p>
              </Field>

              <Field label="Target ICPs" icon="🎯">
                <div className="flex gap-1.5 flex-wrap">
                  {DEFAULT_ICPS.map(icp => (
                    <button key={icp} onClick={() => toggleICP(icp)}
                      className={`text-[10px] px-2 py-1 rounded-md font-medium transition-all ${
                        (form.target_icps || []).includes(icp)
                          ? 'bg-np-blue text-white'
                          : 'bg-gray-50 text-gray-500 hover:bg-gray-100'
                      }`}>
                      {icp}
                    </button>
                  ))}
                </div>
              </Field>

              {/* ─── Host & Show Info ─── */}
              <Section title="Host & Show Info" icon={<User className="w-4 h-4 text-amber-500" />} defaultOpen>
                <div className="grid grid-cols-2 gap-3">
                  <Field label="Host Name">
                    <input value={form.host_name || ''} onChange={e => updateForm('host_name', e.target.value)}
                      placeholder="Host full name" className={inputCls} />
                  </Field>
                  <Field label="Host Email/Contact">
                    <input value={form.host_email || ''} onChange={e => updateForm('host_email', e.target.value)}
                      placeholder="email or handle" className={inputCls} />
                  </Field>
                </div>
                <Field label="What the Host is Excited About" icon="🔥">
                  <div className="relative">
                    <textarea value={form.host_excited_about || ''} onChange={e => updateForm('host_excited_about', e.target.value)}
                      rows={2} className={textareaCls} />
                    <button onClick={() => isRecording && recordingField === 'host_excited_about' ? stopRecording() : startRecording('host_excited_about')}
                      className="absolute right-2 top-2 p-1 rounded hover:bg-gray-100">
                      {isRecording && recordingField === 'host_excited_about'
                        ? <Pause className="w-3 h-3 text-red-500" />
                        : <Volume2 className="w-3 h-3 text-gray-300" />}
                    </button>
                  </div>
                </Field>
                <Field label="What the Host Cares About" icon="🎯">
                  <textarea value={form.host_cares_about || ''} onChange={e => updateForm('host_cares_about', e.target.value)}
                    rows={2} className={textareaCls} />
                </Field>
                <Field label="Interview Style" icon="🎙️">
                  <textarea value={form.interview_style || ''} onChange={e => updateForm('interview_style', e.target.value)}
                    placeholder="Conversational? Inquisitive? Story-driven?" rows={2} className={textareaCls} />
                </Field>
                <Field label="Show Website">
                  <input value={form.show_website || ''} onChange={e => updateForm('show_website', e.target.value)}
                    placeholder="https://..." className={inputCls} />
                </Field>
                <Field label="Audience / Listenership">
                  <input value={form.audience_info || ''} onChange={e => updateForm('audience_info', e.target.value)}
                    placeholder="e.g. 50K monthly listeners, health/wellness niche" className={inputCls} />
                </Field>
                <Field label="Show Notes / Background">
                  <textarea value={form.show_notes || ''} onChange={e => updateForm('show_notes', e.target.value)}
                    placeholder="Background on the show, recent episodes, host's style, past guests..."
                    rows={3} className={textareaCls} />
                </Field>
              </Section>

              {/* ─── Prep Sheet Upload ─── */}
              <Section title="Import Prep Sheet" icon={<Upload className="w-4 h-4 text-blue-500" />}>
                <div
                  onDragOver={e => { e.preventDefault(); setPrepSheetDragging(true) }}
                  onDragLeave={() => setPrepSheetDragging(false)}
                  onDrop={e => { e.preventDefault(); setPrepSheetDragging(false); const f = e.dataTransfer.files[0]; if (f) handleFileUpload(f) }}
                  className={`border-2 border-dashed rounded-xl p-6 text-center transition-colors ${
                    prepSheetDragging ? 'border-np-blue bg-np-blue/5' : 'border-gray-200'
                  }`}>
                  {parsingPrepSheet ? (
                    <div className="flex items-center justify-center gap-2">
                      <Loader2 className="w-5 h-5 animate-spin text-np-blue" />
                      <span className="text-xs text-gray-500">AI is reading your prep sheet...</span>
                    </div>
                  ) : (
                    <>
                      <Upload className="w-8 h-8 text-gray-300 mx-auto mb-2" />
                      <p className="text-xs text-gray-500 mb-2">Drag & drop prep sheet or pitch doc</p>
                      <button onClick={() => fileInputRef.current?.click()}
                        className="text-[10px] px-3 py-1.5 bg-gray-100 rounded-lg text-gray-600 hover:bg-gray-200">
                        Choose File
                      </button>
                      <input ref={fileInputRef} type="file" accept=".txt,.md,.csv,.doc" className="hidden"
                        onChange={e => { const f = e.target.files?.[0]; if (f) handleFileUpload(f) }} />
                    </>
                  )}
                </div>
                <p className="text-[9px] text-gray-300 mt-2">AI will extract podcast name, host info, questions, and talking points automatically</p>
              </Section>

              {/* ─── Strategic Positioning ─── */}
              <Section title="Strategic Positioning" icon={<Target className="w-4 h-4 text-purple-500" />}>
                <Field label="Positioning Notes">
                  <div className="relative">
                    <textarea value={form.strategic_positioning || ''} onChange={e => updateForm('strategic_positioning', e.target.value)}
                      placeholder="How to position yourself for this specific audience..." rows={4} className={textareaCls} />
                    <button onClick={() => isRecording && recordingField === 'strategic_positioning' ? stopRecording() : startRecording('strategic_positioning')}
                      className="absolute right-2 top-2 p-1 rounded hover:bg-gray-100">
                      {isRecording && recordingField === 'strategic_positioning'
                        ? <Pause className="w-3 h-3 text-red-500" />
                        : <Volume2 className="w-3 h-3 text-gray-300" />}
                    </button>
                  </div>
                </Field>
              </Section>

              {/* ─── Talking Points & Prep ─── */}
              <Section title="Talking Points & Prep" icon={<FileText className="w-4 h-4 text-blue-500" />} defaultOpen>
                {/* Pull from Future Ideas */}
                {futureIdeas.length > 0 && (
                  <div className="mb-3 p-2 bg-amber-50 rounded-lg border border-amber-100">
                    <p className="text-[9px] font-bold text-amber-700 mb-1">📌 Pull from Future Ideas</p>
                    <select onChange={e => {
                      const idea = futureIdeas.find(i => i.id === e.target.value)
                      if (idea) {
                        updateForm('key_talking_points', (form.key_talking_points || '') + '\n• ' + idea.title)
                      }
                      e.target.value = ''
                    }} className={selectCls}>
                      <option value="">Select a future idea to link...</option>
                      {futureIdeas.map(i => <option key={i.id} value={i.id}>📌 {i.title}</option>)}
                    </select>
                  </div>
                )}
                <Field label="Key Messages / Talking Points">
                  <div className="relative">
                    <textarea value={form.key_talking_points || ''} onChange={e => updateForm('key_talking_points', e.target.value)}
                      rows={5} className={textareaCls} placeholder="Key points to prepare..." />
                    <button onClick={() => isRecording && recordingField === 'key_talking_points' ? stopRecording() : startRecording('key_talking_points')}
                      className="absolute right-2 top-2 p-1 rounded hover:bg-gray-100">
                      {isRecording && recordingField === 'key_talking_points'
                        ? <Pause className="w-3 h-3 text-red-500" />
                        : <Volume2 className="w-3 h-3 text-gray-300" />}
                    </button>
                  </div>
                </Field>
                <Field label="Stories / Anecdotes to Tell">
                  <textarea value={form.stories_anecdotes || ''} onChange={e => updateForm('stories_anecdotes', e.target.value)}
                    rows={4} className={textareaCls} placeholder="Personal stories and examples..." />
                </Field>
                <Field label="CTA / Offer to Mention">
                  <input value={form.cta_offer || ''} onChange={e => updateForm('cta_offer', e.target.value)}
                    placeholder="e.g. Free quiz at neuroprogeny.com/quiz" className={inputCls} />
                </Field>
              </Section>

              {/* ─── Questions to Expect ─── */}
              <Section title="Questions to Expect" icon={<MessageSquare className="w-4 h-4 text-rose-500" />}
                badge={questions.length}>
                <div className="flex gap-2 mb-3">
                  <input value={newQuestion} onChange={e => setNewQuestion(e.target.value)}
                    onKeyDown={e => { if (e.key === 'Enter') handleAddQuestion() }}
                    placeholder="Add a question..." className={`${inputCls} flex-1`} />
                  <button onClick={handleAddQuestion}
                    className="px-3 py-2 bg-gray-100 rounded-lg text-[10px] font-medium text-gray-600 hover:bg-gray-200">Add</button>
                </div>
                {questions.length > 0 && (
                  <button onClick={handleDraftAnswers} disabled={aiDrafting}
                    className="flex items-center gap-1.5 px-3 py-1.5 bg-np-blue/10 text-np-blue rounded-lg text-[10px] font-medium mb-3 hover:bg-np-blue/20 disabled:opacity-50">
                    {aiDrafting ? <Loader2 className="w-3 h-3 animate-spin" /> : <Sparkles className="w-3 h-3" />}
                    AI Draft All Answers
                  </button>
                )}
                <div className="space-y-2">
                  {questions.map((q, i) => (
                    <div key={q.id} className="border border-gray-100 rounded-lg p-3 group">
                      <div className="flex items-start gap-2">
                        <span className="text-[10px] font-bold text-gray-400 mt-0.5">{i + 1}.</span>
                        <div className="flex-1">
                          <p className="text-xs font-medium text-np-dark mb-1">{q.question}</p>
                          {q.draft_answer ? (
                            <textarea value={q.draft_answer}
                              onChange={e => {
                                setQuestions(prev => prev.map(qq => qq.id === q.id ? { ...qq, draft_answer: e.target.value } : qq))
                                updateQuestion(q.id, { draft_answer: e.target.value })
                              }}
                              rows={3} className={`${textareaCls} bg-green-50/50 border-green-200`} />
                          ) : (
                            <p className="text-[10px] text-gray-300 italic">No answer drafted yet</p>
                          )}
                        </div>
                        <button onClick={() => { deleteQuestion(q.id); setQuestions(prev => prev.filter(qq => qq.id !== q.id)) }}
                          className="opacity-0 group-hover:opacity-100 p-1 hover:bg-red-50 rounded">
                          <Trash2 className="w-3 h-3 text-gray-300 hover:text-red-500" />
                        </button>
                      </div>
                    </div>
                  ))}
                </div>
              </Section>

              {/* ─── Feedback Coach Panel ─── */}
              <Section title="Feedback Coach" icon={<Users className="w-4 h-4 text-indigo-500" />}>
                <p className="text-[10px] text-gray-400 mb-3">
                  Get feedback from your advisory panel. Active voices will critique your talking points, CTA, and positioning.
                </p>
                <div className="flex gap-1.5 flex-wrap mb-3">
                  {advisoryVoices.map(v => (
                    <button key={v.id} onClick={() => toggleVoice(v.id, !v.active)}
                      className={`text-[10px] px-2.5 py-1 rounded-full font-medium border-2 transition-all ${
                        v.active ? 'text-white border-transparent' : 'bg-white border-gray-200 text-gray-400'
                      }`}
                      style={v.active ? { backgroundColor: v.color, borderColor: v.color } : {}}>
                      {v.name}
                    </button>
                  ))}
                </div>
                <button onClick={handleFeedbackCoach} disabled={feedbackLoading || advisoryVoices.filter(v => v.active).length === 0}
                  className="flex items-center gap-1.5 px-4 py-2 bg-gradient-to-r from-indigo-500 to-purple-500 text-white rounded-lg text-xs font-medium hover:opacity-90 disabled:opacity-50 mb-3">
                  {feedbackLoading ? <Loader2 className="w-3.5 h-3.5 animate-spin" /> : <Zap className="w-3.5 h-3.5" />}
                  Get Feedback
                </button>
                {feedbackResults && (
                  <div className="space-y-3">
                    <div className="flex items-center gap-2 p-2 bg-gray-50 rounded-lg">
                      <span className="text-xs font-bold text-np-dark">Overall Score:</span>
                      <span className="text-lg font-bold text-np-blue">{feedbackResults.overall_score}/10</span>
                      <span className="text-[10px] text-gray-500 ml-2">{feedbackResults.top_priority}</span>
                    </div>
                    {feedbackResults.feedback?.map((fb: any, i: number) => (
                      <div key={i} className="border-l-4 rounded-lg p-3 bg-white" style={{ borderColor: fb.voice_color }}>
                        <div className="flex items-center gap-2 mb-2">
                          <span className="text-xs font-bold" style={{ color: fb.voice_color }}>{fb.voice_name}</span>
                          <span className="text-[10px] text-gray-400">{fb.rating}/10</span>
                        </div>
                        {fb.key_quote && (
                          <p className="text-xs text-gray-600 italic mb-2">"{fb.key_quote}"</p>
                        )}
                        {fb.strengths?.map((s: string, j: number) => (
                          <p key={j} className="text-[10px] text-green-600 mb-0.5">✓ {s}</p>
                        ))}
                        {fb.suggestions?.map((s: string, j: number) => (
                          <p key={j} className="text-[10px] text-amber-600 mb-0.5">→ {s}</p>
                        ))}
                      </div>
                    ))}
                  </div>
                )}
              </Section>

              {/* ─── Prep Checklist ─── */}
              {editingPodcast && (
                <Section title="Preparation Checklist" icon={<CheckSquare className="w-4 h-4 text-green-500" />}
                  badge={`${checklist.filter(c => c.completed).length}/${checklist.length}`}>
                  <div className="space-y-1 mb-2">
                    {checklist.map(item => (
                      <label key={item.id} className="flex items-center gap-2 px-2 py-1.5 rounded-lg hover:bg-gray-50 cursor-pointer">
                        <button onClick={() => {
                          toggleChecklistItem(item.id, !item.completed)
                          setChecklist(prev => prev.map(c => c.id === item.id ? { ...c, completed: !c.completed } : c))
                        }}>
                          {item.completed
                            ? <CheckSquare className="w-4 h-4 text-green-500" />
                            : <Square className="w-4 h-4 text-gray-300" />}
                        </button>
                        <span className={`text-xs ${item.completed ? 'line-through text-gray-400' : 'text-np-dark'}`}>{item.label}</span>
                      </label>
                    ))}
                  </div>
                  <div className="flex gap-2">
                    <input value={newCheckItem} onChange={e => setNewCheckItem(e.target.value)}
                      onKeyDown={async e => {
                        if (e.key === 'Enter' && newCheckItem.trim()) {
                          const item = await addChecklistItem(editingPodcast.id, newCheckItem.trim())
                          if (item) setChecklist(prev => [...prev, item])
                          setNewCheckItem('')
                        }
                      }}
                      placeholder="Add checklist item..." className={`${inputCls} flex-1`} />
                  </div>
                </Section>
              )}

              {/* ─── Technical & Logistics ─── */}
              <Section title="Technical & Logistics" icon={<Settings2 className="w-4 h-4 text-gray-400" />}>
                <Field label="Technical Notes">
                  <textarea value={form.tech_notes || ''} onChange={e => updateForm('tech_notes', e.target.value)}
                    placeholder="Audio setup, lighting, background..." rows={2} className={textareaCls} />
                </Field>
                <Field label="Recording Link">
                  <input value={form.recording_link || ''} onChange={e => updateForm('recording_link', e.target.value)}
                    placeholder="Zoom/Riverside link" className={inputCls} />
                </Field>
              </Section>

              {/* ─── Post-Episode Social ─── */}
              <Section title="Post-Episode Social" icon={<TrendingUp className="w-4 h-4 text-pink-500" />}>
                <button onClick={handleGenerateSocial} disabled={socialLoading}
                  className="flex items-center gap-1.5 px-3 py-1.5 bg-pink-50 text-pink-600 rounded-lg text-[10px] font-medium mb-3 hover:bg-pink-100 disabled:opacity-50">
                  {socialLoading ? <Loader2 className="w-3 h-3 animate-spin" /> : <Sparkles className="w-3 h-3" />}
                  Generate 6 Social Posts
                </button>
                {socialPosts.length > 0 && (
                  <div className="space-y-2">
                    {socialPosts.map((post, i) => (
                      <div key={i} className="border border-gray-100 rounded-lg p-3">
                        <div className="flex items-center gap-2 mb-1">
                          <span className="text-[9px] font-bold px-1.5 py-0.5 rounded bg-gray-100 text-gray-600">{post.platform}</span>
                          <span className="text-[9px] text-gray-400">{post.type}</span>
                          <button onClick={() => navigator.clipboard.writeText(post.content)}
                            className="ml-auto p-1 hover:bg-gray-100 rounded">
                            <Copy className="w-3 h-3 text-gray-400" />
                          </button>
                        </div>
                        <p className="text-xs text-np-dark whitespace-pre-wrap">{post.content}</p>
                      </div>
                    ))}
                  </div>
                )}
                <Field label="Additional Social Notes">
                  <textarea value={form.post_social_notes || ''} onChange={e => updateForm('post_social_notes', e.target.value)}
                    rows={2} className={textareaCls} placeholder="Clips to pull, quotes to highlight..." />
                </Field>
              </Section>

              {/* ─── Retrospect ─── */}
              <Section title="Retrospect" icon={<Lightbulb className="w-4 h-4 text-amber-500" />}>
                <Field label="What Went Well">
                  <textarea value={form.retro_went_well || ''} onChange={e => updateForm('retro_went_well', e.target.value)}
                    rows={3} className={textareaCls} />
                </Field>
                <Field label="What to Improve">
                  <textarea value={form.retro_improve || ''} onChange={e => updateForm('retro_improve', e.target.value)}
                    rows={3} className={textareaCls} />
                </Field>
                <Field label="Episode Rating">
                  <div className="flex gap-1">
                    {[1, 2, 3, 4, 5].map(n => (
                      <button key={n} onClick={() => updateForm('retro_rating', n)}
                        className="p-1 hover:scale-110 transition-transform">
                        <Star className={`w-5 h-5 ${(form.retro_rating || 0) >= n ? 'text-amber-400 fill-amber-400' : 'text-gray-200'}`} />
                      </button>
                    ))}
                  </div>
                </Field>
                <Field label="Capture Topics for Future Podcasts" icon="📌">
                  <p className="text-[9px] text-gray-300 mb-1">Topics that came up worth exploring in future appearances</p>
                  <div className="flex gap-1.5 flex-wrap mb-2">
                    {(form.retro_topics_captured || []).map((t, i) => (
                      <span key={i} className="text-[10px] px-2 py-1 rounded-md bg-amber-50 text-amber-700 flex items-center gap-1">
                        📌 {t}
                        <button onClick={() => updateForm('retro_topics_captured', (form.retro_topics_captured || []).filter((_, j) => j !== i))}>
                          <X className="w-2.5 h-2.5" />
                        </button>
                      </span>
                    ))}
                  </div>
                  <div className="flex gap-2">
                    <input placeholder="New topic..."
                      onKeyDown={e => {
                        if (e.key === 'Enter') {
                          const val = (e.target as HTMLInputElement).value.trim()
                          if (val) {
                            updateForm('retro_topics_captured', [...(form.retro_topics_captured || []), val])
                            addFutureIdea(val, undefined, editingPodcast?.id)
                            ;(e.target as HTMLInputElement).value = ''
                          }
                        }
                      }}
                      className={`${inputCls} flex-1`} />
                  </div>
                </Field>
              </Section>

            </div>

            {/* Modal footer */}
            <div className="flex items-center gap-3 px-6 py-4 border-t border-gray-100 sticky bottom-0 bg-white rounded-b-2xl">
              {editingPodcast && (
                <button onClick={async () => {
                  if (confirm('Delete this podcast?')) {
                    await deletePodcast(editingPodcast.id)
                    setModalOpen(false)
                  }
                }} className="flex items-center gap-1.5 px-3 py-2 text-red-500 hover:bg-red-50 rounded-lg text-xs font-medium">
                  <Trash2 className="w-3.5 h-3.5" /> Delete
                </button>
              )}
              <div className="flex-1" />
              <button onClick={() => setModalOpen(false)}
                className="px-4 py-2 text-xs font-medium text-gray-500 hover:bg-gray-50 rounded-lg">Cancel</button>
              <button onClick={handleSave} disabled={saving || !form.name?.trim()}
                className="flex items-center gap-1.5 px-4 py-2 bg-np-blue text-white rounded-lg text-xs font-medium hover:bg-np-blue/90 disabled:opacity-50">
                {saving ? <Loader2 className="w-3.5 h-3.5 animate-spin" /> : null}
                {editingPodcast ? 'Save Podcast' : 'Create Podcast'}
              </button>
            </div>
          </div>
        </div>
      )}

      {/* ═══════════════════════════════════════════
          VOICE PANEL MODAL
          ═══════════════════════════════════════════ */}
      {voicePanelOpen && (
        <div className="fixed inset-0 bg-black/40 z-50 flex items-center justify-center">
          <div className="bg-white rounded-2xl w-full max-w-md mx-4 shadow-2xl">
            <div className="flex items-center gap-3 px-6 py-4 border-b border-gray-100">
              <Users className="w-5 h-5 text-indigo-500" />
              <h2 className="text-sm font-bold text-np-dark flex-1">Advisory Voice Library</h2>
              <button onClick={() => setVoicePanelOpen(false)} className="p-1.5 hover:bg-gray-100 rounded-lg">
                <X className="w-4 h-4 text-gray-400" />
              </button>
            </div>
            <div className="px-6 py-4 max-h-[60vh] overflow-y-auto">
              <p className="text-[10px] text-gray-400 mb-3">
                Toggle voices on/off. Active voices will provide feedback during the Feedback Coach panel.
              </p>
              <div className="space-y-2 mb-4">
                {advisoryVoices.map(v => (
                  <div key={v.id} className="flex items-start gap-3 p-3 border border-gray-100 rounded-xl group">
                    <button onClick={() => toggleVoice(v.id, !v.active)}
                      className={`w-8 h-8 rounded-full flex items-center justify-center text-[10px] font-bold text-white flex-shrink-0 mt-0.5 ${
                        v.active ? '' : 'opacity-30'
                      }`} style={{ backgroundColor: v.color }}>
                      {v.name[0]}
                    </button>
                    <div className="flex-1 min-w-0">
                      <div className="flex items-center gap-2">
                        <span className="text-xs font-semibold text-np-dark">{v.name}</span>
                        {v.role && <span className="text-[9px] text-gray-400">{v.role}</span>}
                      </div>
                      <p className="text-[10px] text-gray-500 mt-0.5 line-clamp-2">{v.perspective}</p>
                    </div>
                    {!v.is_default && (
                      <button onClick={() => deleteVoice(v.id)}
                        className="opacity-0 group-hover:opacity-100 p-1 hover:bg-red-50 rounded">
                        <Trash2 className="w-3 h-3 text-gray-300 hover:text-red-500" />
                      </button>
                    )}
                  </div>
                ))}
              </div>
              {/* Add new voice */}
              <div className="border border-gray-200 rounded-xl p-3">
                <p className="text-[10px] font-bold text-gray-400 uppercase tracking-wider mb-2">Add New Voice</p>
                <input value={newVoiceName} onChange={e => setNewVoiceName(e.target.value)}
                  placeholder="Name (e.g. Gary Vee)" className={`${inputCls} mb-2`} />
                <input value={newVoiceRole} onChange={e => setNewVoiceRole(e.target.value)}
                  placeholder="Role (e.g. Content & Distribution)" className={`${inputCls} mb-2`} />
                <textarea value={newVoicePerspective} onChange={e => setNewVoicePerspective(e.target.value)}
                  placeholder="How this voice thinks, what they push for, their framework..."
                  rows={3} className={`${textareaCls} mb-2`} />
                <button onClick={async () => {
                  if (!newVoiceName.trim() || !newVoicePerspective.trim()) return
                  await addVoice({
                    name: newVoiceName.trim(), role: newVoiceRole.trim() || null,
                    perspective: newVoicePerspective.trim(), color: '#' + Math.floor(Math.random()*16777215).toString(16),
                    active: true, sort_order: advisoryVoices.length,
                  })
                  setNewVoiceName(''); setNewVoiceRole(''); setNewVoicePerspective('')
                }} disabled={!newVoiceName.trim() || !newVoicePerspective.trim()}
                  className="w-full px-3 py-2 bg-np-blue text-white rounded-lg text-xs font-medium disabled:opacity-50">
                  Add Voice
                </button>
              </div>
            </div>
          </div>
        </div>
      )}
    </div>
  )
}
